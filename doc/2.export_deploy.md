# Instructions for exporting the project from the development environment to the production environment


``` bash
    @echo off
    REM Cria um ambiente virtual
    python -m venv myenv
    call myenv\Scripts\activate

    REM Instala as dependências
    pip install -r requirements.txt

    REM Baixa todas as dependências para a pasta 'packages'
    pip download -r requirements.txt -d packages

    REM Empacota o projeto
    pip install build
    python -m build

    REM Cria a pasta de deploy
    mkdir deploy

    REM Copia o ambiente virtual e as dependências para a pasta 'deploy'
    xcopy myenv deploy\myenv /E /H /C /I /Y
    xcopy packages deploy\packages /E /H /C /I /Y
    xcopy dist deploy\dist /E /H /C /I /Y

    REM Copia os arquivos do projeto para a pasta 'deploy', excluindo os arquivos listados em 'deploy_exclude.txt'
    xcopy . deploy /E /H /C /I /Y /EXCLUDE:deploy_exclude.txt

    REM Cria um arquivo ZIP contendo a pasta de deploy
    powershell Compress-Archive -Path deploy -DestinationPath deploy.zip

    REM Desativa o ambiente virtual
    deactivate

    echo Deploy Export concluído com sucesso!
    pause
```
